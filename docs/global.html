Source:

Example
{
  'bitcoin': {
    'usd': {
      'url': 'https://api.coingecko.com/api/v3/simple/price?ids=bitcoin&vs_currencies=usd',
      'transform': function (response) {
        if (response && response.bitcoin && response.bitcoin.usd) {
          return response.bitcoin.usd
        }
      }
    }
  }
}
